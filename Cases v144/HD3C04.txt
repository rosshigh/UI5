sap.ui.define([
   "sap/ui/core/mvc/Controller",
   "sap/m/MessageToast"
], function (Controller, MessageToast) {
   "use strict";
   return Controller.extend("ui5.controller.App", {
       doIt : function(oEvent) {
            MessageToast.show("Pressed: " + oEvent.getSource().getId());
        }
   });
});
